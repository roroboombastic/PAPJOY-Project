function validateEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

async function hashPassword(password) {
  return bcrypt.hash(password, 10);
}

async function comparePasswords(password, hash) {
  return bcrypt.compare(password, hash);
}

async function getWebhookStatus() {
    let status = await WebhookStatus.findOne().sort({ updatedAt: -1 });
    if (!status) {
        status = await WebhookStatus.create({ lastEvent: null, lastReceivedAt: null, lastError: null, count: 0 });
    }
    return status;
}

async function saveOrder({ provider, products, amount, paymentId, status, userId = null, deliveryInfo = null, paymentMethod = 'web', codNotes = null, cardInfo = null, orderId = null }) {
    if (!orderId) {
        orderId = `PJ-${Date.now().toString().slice(-4)}-${Math.floor(1000 + Math.random() * 9000)}`;
    }

    const existingOrder = await Order.findOne({ provider, paymentId });
    if (existingOrder) {
        return existingOrder;
    }

    const order = await Order.create({
        orderId,
        provider,
        userId,
        products,
        amount,
        paymentId,
        status,
        deliveryInfo,
        paymentMethod,
        codNotes,
        cardInfo,
    });
    return order;
}

app.post('/api/stripe-webhook', bodyParser.raw({ type: 'application/json' }), async (req, res) => {
    const signature = req.headers['stripe-signature'];
    const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET;

    if (!webhookSecret) {
        return res.status(500).send('Stripe webhook secret is not configured.');
    }

    let event;
    try {
        event = stripe.webhooks.constructEvent(req.body, signature, webhookSecret);
    } catch (error) {
        console.error('Stripe webhook verification failed:', error.message);
        return res.status(400).send(`Webhook Error: ${error.message}`);
    }

    const status = await getWebhookStatus();
    status.lastEvent = event.type;
    status.lastReceivedAt = new Date();
    status.lastError = null;
    status.count += 1;
    await status.save();

    if (event.type === 'checkout.session.completed') {
        const session = event.data.object;
        try {
            const lineItems = await stripe.checkout.sessions.listLineItems(session.id, { limit: 100 });
            const productsForOrder = lineItems.data.map((item) => ({
                name: item.description || 'PAPJOY item',
                quantity: item.quantity,
                price: (item.amount_total || 0) / 100,
            }));

            saveOrder({
                provider: 'Stripe',
                products: productsForOrder,
                amount: session.amount_total / 100,
                paymentId: session.payment_intent,
                status: 'Paid',
            });
        } catch (saveError) {
            console.error('Unable to save Stripe order from webhook:', saveError);
        }
    }

    res.json({ received: true });
});

app.post('/api/auth/register', async (req, res) => {
  const { email, password, name } = req.body;
  if (!email || !password) {
    return res.status(400).json({ error: 'Email and password are required.' });
  }
  if (!validateEmail(email)) {
    return res.status(400).json({ error: 'Please provide a valid email address.' });
  }
  if (password.length < 8) {
    return res.status(400).json({ error: 'Password must be at least 8 characters long.' });
  }

  try {
    const existing = await User.findOne({ email: email.toLowerCase() });
    if (existing) {
      return res.status(409).json({ error: 'Email is already registered.' });
    }

    const passwordHash = await hashPassword(password);
    const user = await User.create({
      email: email.toLowerCase(),
      passwordHash,
      name: name || 'PAP-JOY Member',
      shippingAddress: {
        fullName: req.body.shippingAddress?.fullName || '',
        line1: req.body.shippingAddress?.line1 || '',
        line2: req.body.shippingAddress?.line2 || '',
        city: req.body.shippingAddress?.city || '',
        state: req.body.shippingAddress?.state || '',
        postalCode: req.body.shippingAddress?.postalCode || '',
        country: req.body.shippingAddress?.country || '',
        phone: req.body.shippingAddress?.phone || '',
      },
    });

    res.status(201).json({ email: user.email, name: user.name, id: user._id, cart: user.cart || [] });
  } catch (error) {
    console.error('Auth registration error', error);
    res.status(500).json({ error: 'Unable to register user.' });
  }
});

app.post('/api/auth/login', async (req, res) => {
  const { email, password } = req.body;
  if (!email || !password) {
    return res.status(400).json({ error: 'Email and password are required.' });
  }

  try {
    const user = await User.findOne({ email: email.toLowerCase() });
    if (!user) {
      return res.status(401).json({ error: 'Invalid email or password.' });
    }

    const valid = await comparePasswords(password, user.passwordHash);
    if (!valid) {
      return res.status(401).json({ error: 'Invalid email or password.' });
    }

    res.json({ email: user.email, name: user.name, id: user._id, cart: user.cart || [] });
  } catch (error) {
    console.error('Auth login error', error);
    res.status(500).json({ error: 'Unable to process login.' });
  }
});

app.put('/api/auth/account/:id', async (req, res) => {
  const { id } = req.params;
  const { name, shippingAddress, deliveryPreferences } = req.body;

  if (!id) {
    return res.status(400).json({ error: 'Account ID is required.' });
  }

  try {
    const user = await User.findById(id);
    if (!user) {
      return res.status(404).json({ error: 'Account not found.' });
    }

    if (name) user.name = name;
    if (shippingAddress && typeof shippingAddress === 'object') {
      user.shippingAddress = {
        fullName: shippingAddress.fullName || user.shippingAddress.fullName,
        line1: shippingAddress.line1 || user.shippingAddress.line1,
        line2: shippingAddress.line2 || user.shippingAddress.line2,
        city: shippingAddress.city || user.shippingAddress.city,
        state: shippingAddress.state || user.shippingAddress.state,
        postalCode: shippingAddress.postalCode || user.shippingAddress.postalCode,
        country: shippingAddress.country || user.shippingAddress.country,
        phone: shippingAddress.phone || user.shippingAddress.phone,
      };
    }
    
    if (deliveryPreferences && typeof deliveryPreferences === 'object') {
      user.deliveryPreferences = {
        preferredLocation: deliveryPreferences.preferredLocation || user.deliveryPreferences.preferredLocation,
        instructions: deliveryPreferences.instructions || user.deliveryPreferences.instructions,
        weekdayDelivery: deliveryPreferences.weekdayDelivery !== undefined ? deliveryPreferences.weekdayDelivery : user.deliveryPreferences.weekdayDelivery,
        weekendDelivery: deliveryPreferences.weekendDelivery !== undefined ? deliveryPreferences.weekendDelivery : user.deliveryPreferences.weekendDelivery,
      };
    }

    const { passwordHash, __v, ...publicUser } = user.toObject();
    res.json(publicUser);

    console.error('Auth account update error', error);
    res.status(500).json({ error: 'Unable to update account details.' });
  }

// Create generic order from cart
app.post("/api/orders", async (req, res) => {
    const { items, userId, paymentMethod, deliveryInfo, codNotes, cardInfo } = req.body;
    if (!Array.isArray(items) || items.length === 0) {
        return res.status(400).json({ error: 'Cart is empty.' });
    }

    const amount = items.reduce((sum, item) => sum + (item.price || 0) * (item.quantity || 1), 0);
    const productsPayload = items.map((item) => ({
        name: item.name,
        price: item.price,
        quantity: item.quantity || 1,
        category: item.category,
        subtitle: item.subtitle,
    }));

    // Add COD fee if applicable
    let finalAmount = amount;
    if (paymentMethod === 'cod') {
        finalAmount += 50; // COD fee
    }

    try {
        const order = await saveOrder({
            orderId: `PJ-${Date.now().toString().slice(-4)}-${Math.floor(1000 + Math.random() * 9000)}`,
            provider: paymentMethod === 'cod' ? 'COD' : paymentMethod === 'creditcard' ? 'Credit Card' : paymentMethod === 'debitcard' ? 'Debit Card' : 'Web',
            userId: userId || null,
            products: productsPayload,
            amount: finalAmount,
            paymentId: `${paymentMethod?.toUpperCase() || 'WEB'}-${Date.now()}`,
            status: paymentMethod === 'cod' ? 'pending' : 'paid',
            deliveryInfo: deliveryInfo || null,
            paymentMethod: paymentMethod || 'web',
            codNotes: codNotes || null,
            cardInfo: cardInfo ? {
                number: cardInfo.number ? cardInfo.number.slice(-4) : null, // Only store last 4 digits
                expiry: cardInfo.expiry || null,
                name: cardInfo.name || null
            } : null,
        });

        res.json({ order });
    } catch (error) {
        console.error('Order save error', error);
        res.status(500).json({ error: 'Unable to save order.' });
    }
});

function createPayPalClient() {
    const clientId = process.env.PAYPAL_CLIENT_ID;
    const clientSecret = process.env.PAYPAL_CLIENT_SECRET || process.env.PAYPAL_SECRET;
    const mode = process.env.PAYPAL_MODE === 'live' ? 'live' : 'sandbox';

    const env = mode === 'live'
        ? new paypal.core.LiveEnvironment(clientId, clientSecret)
        : new paypal.core.SandboxEnvironment(clientId, clientSecret);
    return new paypal.core.PayPalHttpClient(env);
}

app.post('/api/stripe-session', async (req, res) => {
    if (!process.env.STRIPE_SECRET_KEY) {
        return res.status(500).json({ error: 'Stripe is not configured. Missing STRIPE_SECRET_KEY.' });
    }

    const { items } = req.body;
    if (!Array.isArray(items) || items.length === 0) {
        return res.status(400).json({ error: 'Cart is empty.' });
    }

    const line_items = items.map((product) => ({
        price_data: {
            currency: 'INR',
            product_data: {
                name: product.name,
                description: product.subtitle || 'PAPJOY footwear',
            },
            unit_amount: Math.round(product.price * 100),
        },
        quantity: product.quantity,
    }));

    try {
        const session = await stripe.checkout.sessions.create({
            payment_method_types: ['card'],
            mode: 'payment',
            line_items,
            success_url: process.env.STRIPE_SUCCESS_URL || `${APP_URL}/success.html?provider=stripe&session_id={CHECKOUT_SESSION_ID}`,
            cancel_url: process.env.STRIPE_CANCEL_URL || `${APP_URL}/checkout.html?checkout=canceled`,
        });
        res.json({ url: session.url });
    } catch (error) {
        console.error('Stripe session error', error);
        res.status(500).json({ error: 'Unable to create Stripe checkout session.' });
    }
});

app.post('/api/stripe-order', async (req, res) => {
    if (!process.env.STRIPE_SECRET_KEY) {
        return res.status(500).json({ error: 'Stripe is not configured. Missing STRIPE_SECRET_KEY.' });
    }

    const { sessionId, items } = req.body;
    if (!sessionId) {
        return res.status(400).json({ error: 'Missing session ID.' });
    }

    try {
        const session = await stripe.checkout.sessions.retrieve(sessionId, {
            expand: ['payment_intent']
        });

        const order = await saveOrder({
            provider: 'Stripe',
            products: items || [],
            amount: session.amount_total / 100,
            paymentId: session.payment_intent,
            status: session.payment_status || 'paid'
        });

        res.json({ order });
    } catch (error) {
        console.error('Stripe order save error', error);
        res.status(500).json({ error: 'Unable to save Stripe order.' });
    }
});

app.post('/api/paypal-order', async (req, res) => {
    if (!process.env.PAYPAL_CLIENT_ID || !process.env.PAYPAL_CLIENT_SECRET) {
        return res.status(500).json({ error: 'PayPal is not configured. Missing PAYPAL_CLIENT_ID or PAYPAL_CLIENT_SECRET.' });
    }

    const { items } = req.body;
    if (!Array.isArray(items) || items.length === 0) {
        return res.status(400).json({ error: 'Cart is empty.' });
    }

    const total = items.reduce((sum, item) => sum + item.price * item.quantity, 0).toFixed(2);
    const request = new paypal.orders.OrdersCreateRequest();
    request.prefer('return=representation');
    request.requestBody({
        intent: 'CAPTURE',
        purchase_units: [
            {
                amount: {
                    currency_code: 'INR',
                    value: total,
                    breakdown: {
                        item_total: {
                            currency_code: 'INR',
                            value: total,
                        },
                    },
                },
                items: items.map((product) => ({
                    name: product.name,
                    unit_amount: {
                        currency_code: 'INR',
                        value: product.price.toFixed(2),
                    },
                    quantity: product.quantity.toString(),
                })),
            },
        ],
        application_context: {
            brand_name: 'PAPJOY',
            landing_page: 'BILLING',
            user_action: 'PAY_NOW',
            return_url: process.env.PAYPAL_SUCCESS_URL || `${APP_URL}/success.html?provider=paypal`,
            cancel_url: process.env.PAYPAL_CANCEL_URL || `${APP_URL}/checkout.html?paypal=canceled`,
        },
    });

    try {
        const order = await createPayPalClient().execute(request);
        const approvalUrl = order.result.links.find((link) => link.rel === 'approve')?.href;
        if (!approvalUrl) {
            throw new Error('Approval URL not found');
        }
        res.json({ approvalUrl, orderId: order.result.id });
    } catch (error) {
        console.error('PayPal order error', error);
        res.status(500).json({ error: 'Unable to create PayPal order.' });
    }
});

app.post('/api/paypal-capture', async (req, res) => {
    if (!process.env.PAYPAL_CLIENT_ID || !process.env.PAYPAL_CLIENT_SECRET) {
        return res.status(500).json({ error: 'PayPal is not configured. Missing PAYPAL_CLIENT_ID or PAYPAL_CLIENT_SECRET.' });
    }

    const { orderId, items } = req.body;
    if (!orderId) {
        return res.status(400).json({ error: 'Missing PayPal order ID.' });
    }

    try {
        const request = new paypal.orders.OrdersCaptureRequest(orderId);
        request.requestBody({});

        const capture = await createPayPalClient().execute(request);
        const captureDetails = capture.result.purchase_units?.[0]?.payments?.captures?.[0];
        const paymentId = captureDetails?.id || orderId;
        const amount = captureDetails ? parseFloat(captureDetails.amount.value) : 0;
        const status = captureDetails?.status || capture.result.status;

        const order = await saveOrder({
            provider: 'PayPal',
            products: items || [],
            amount,
            paymentId,
            status,
        });

        res.json({ order });
    } catch (error) {
        console.error('PayPal capture error', error);
        res.status(500).json({ error: 'Unable to capture PayPal order.' });
    }
});

const jwt = require('jsonwebtoken');

app.post('/api/auth/login', async (req, res) => {
  const { email, password } = req.body;

  try {
    const user = await User.findOne({ email: email.toLowerCase() });
    if (!user) {
      return res.status(401).json({ error: 'Invalid email or password.' });
    }

    const valid = await comparePasswords(password, user.passwordHash);
    if (!valid) {
      return res.status(401).json({ error: 'Invalid email or password.' });
    }

    // ✅ CREATE TOKEN
    const token = jwt.sign(
      { id: user._id },
      process.env.JWT_SECRET,
      { expiresIn: "7d" }
    );

    res.json({
      token,
      user: {
        email: user.email,
        name: user.name,
        id: user._id,
        cart: user.cart || []
      }
    });

  } catch (error) {
    console.error('Auth login error', error);
    res.status(500).json({ error: 'Unable to process login.' });
  }
});
startServer();
